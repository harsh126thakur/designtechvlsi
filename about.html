rules_version = '2'; service cloud.firestore { match /databases/{database}/documents { // ================= ADMIN CHECK FUNCTION ================= function isAdmin() { return exists(/databases/$(database)/documents/admins/$(request.auth.token.email)); } // ================= ADMINS ================= match /admins/{adminId} { allow read: if request.auth != null && isAdmin(); allow write: if false; // nobody can modify admins from frontend } // ================= COURSES ================= match /courses/{courseId} { // anyone can view courses allow read: if true; // only admin can create/update/delete allow write: if request.auth != null && isAdmin(); } // ================= ENQUIRIES ================= match /enquiries/{enquiryId} { // users can submit enquiry (no login needed) allow create: if true; // only admin can read or modify allow read, update, delete: if request.auth != null && isAdmin(); } // ================= USERS ================= match /users/{userId} { // user can read/write their own data allow read, write: if request.auth != null && request.auth.uid == userId; // admin can access all users allow read: if request.auth != null && isAdmin(); } } }
